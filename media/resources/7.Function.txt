def factoral(num):              #3=1*2*3    n=3 fact=1
    fact=1                      #1*2*3*4*5*2 n=5 fact=2
    for i in range(1,num+1):
        fact *=i                               
    return fact
n=3
print(factoral(n))

----------------------recursion function--------------------
def factorial(n):
    if n == 0:
        return 1
    return n * factorial(n - 1)


n = int(input("enter a value"))
print(factorial(n))
