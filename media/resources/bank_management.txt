class bank:
    def _init_(self,accno,name,age,phoneno,address):
        self.accno=accno
        self.name=name
        self.age=age
        self.num=phoneno
        self.address=address
        self.balance=0
        self.depositeamount=[]
        self.witdrawamont=[]
    def showaccount(self):
        print("Accountnum",self.accno)
        print("name:",self.name)
        print("Age:",self.age)
        print("Address:",self.address)
    def deposite(self):
        self.amount=int(input("enter your deposite amount:"))
        self.depositeamount.append(self.amount)
        self.balance=self.balance+self.amount
        print(f"{self.amount} deposite successfully")
    def withdraw(self):
        self.amount=int(input("enter your withdraw amount:"))
        self.witdrawamont.append(self.amount)
        self.balance=self.balance-self.amount
        print(f"{self.amount} withdraw successfully")
    def tran_history(self):
        print("Deposite_history:",self.depositeamount)
        print("withdraw history:",self.witdrawamont)

    def currentbalace(self):
        print(f"your current balance is :{self.balance}")
